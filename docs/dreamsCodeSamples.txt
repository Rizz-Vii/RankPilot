  // E.g., for a Sales Agent, it might include user behavior data, lead score, company info.

  [key: string]: any;

  // The structured output generated by the agent after processing a task.
  // E.g., for a Sales Agent, it could be a personalized email draft.
  // For a Marketing Agent, it might be a content calendar.
  // For a Finance Agent, it could be a financial report.
  status: 'success' | 'failure';
  message: string;
  data?: any; // The actual result of the processing.

interface DecisionOptions {
  // Parameters and context for the agent to make a specific decision.

  [key: string]: any;
}
interface Decision {
  // The structured outcome of an agent's decision-making process.
  action: string; // A description of the action to be taken.
  autonomous: boolean; // True if the decision can be executed without human approval.
  reasoning: string; // A brief explanation or justification for the decision.
  requiredApprovalLevel?: DecisionLevel; // If not autonomous, what level of human approval is needed.
}
interface DataQuery {
  // A structured object defining the data an agent needs to access.
  // E.g., { type: 'user', userId: 'xyz' }, { type: 'metrics', department: 'sales', period: 'daily' }.
  [key: string]: any;
}
// Defines the scope of data access permissions for an agent.
type DataScope = 'read' | 'write' | 'full';

export abstract class BaseAIAgent {
  protected department: AgentType; // The specific department this agent belongs to (e.g., 'SALES_AGENT').
  protected capabilities: string[]; // A list of high-level abilities this agent possesses (e.g., 'lead_scoring', 'content_generation').
  protected dataAccess: DataScope[]; // Defines the read/write permissions this agent has to different data stores.
  protected decisionAuthority: DecisionLevel; // The default decision level for this agent's actions as per the DECISION_MATRIX.
  protected dataOrchestrator: DataOrchestrator; // An instance of the centralized DataOrchestrator for data retrieval.
  protected functions = getFunctions(getApp()); // Firebase Cloud Functions instance for calling backend logic.

  constructor(
    department: AgentType,
    capabilities: string[],
    dataAccess: DataScope[],
    decisionAuthority: DecisionLevel
  ) {
    this.department = department;
    this.capabilities = capabilities;
    this.dataAccess = dataAccess;
    this.decisionAuthority = decisionAuthority;
    this.dataOrchestrator = new DataOrchestrator(); // Initialize data access layer
  }

  // Abstract method: Agents must implement how they process incoming tasks/inputs.
  abstract async processInput(context: BusinessContext): Promise<AgentResponse>;
  // Abstract method: Agents must implement their decision-making logic.
  abstract async makeDecision(options: DecisionOptions): Promise<Decision>;
  // Abstract method: Agents must implement how they generate their departmental metrics report.
  abstract async reportMetrics(): Promise<DepartmentMetrics>;

  // Common agent infrastructure methods:
